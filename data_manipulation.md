But both above is cunfusing, we then use piping instead:

``` r
litters_data = 
  read_csv("./data_import_examples/FAS_litters.csv", col_types = "ccddiiii") %>%
  janitor::clean_names() %>%
  select(-pups_survive) %>%
  mutate(
    wt_gain = gd18_weight - gd0_weight,
    group = tolower(group)
  ) %>%
  filter(!is.na(gd0_weight))

head(litters_data, 10)
```
